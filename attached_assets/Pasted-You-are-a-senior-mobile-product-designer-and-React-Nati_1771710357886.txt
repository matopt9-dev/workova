/**
 * One-time admin bootstrap.
 * Run locally/CI with a service account key.
 * Does NOT change app logic, only sets a claim on an existing user.
 */
import admin from "firebase-admin";

const SERVICE_ACCOUNT_PATH = process.env.FIREBASE_SERVICE_ACCOUNT_PATH; // path to JSON
const ADMIN_EMAIL = process.env.ADMIN_EMAIL || "ADMIN_EMAIL_HERE";      // <-- replace

if (!SERVICE_ACCOUNT_PATH) {
  throw new Error("Missing FIREBASE_SERVICE_ACCOUNT_PATH env var");
}

admin.initializeApp({
  credential: admin.credential.cert(require(SERVICE_ACCOUNT_PATH)),
});

async function run() {
  const user = await admin.auth().getUserByEmail(ADMIN_EMAIL);
  await admin.auth().setCustomUserClaims(user.uid, { admin: true });
  console.log(`✅ Set admin=true for ${ADMIN_EMAIL} (${user.uid})`);
  process.exit(0);
}

run().catch((e) => {
  console.error("❌ Failed to set admin claim:", e);
  process.exit(1);
});
